 22 ({
 23     require: ["io", "com", "logs", "theme", "commands", "less"]
 24     ,
 25     loadModule: function ()
 26     {
 27         this.commands.registerCommand("ioctrl", this);
 28     }
 29     ,
 30     ioctrl:
 31     {
 32         desc: "Controls I/O Layer",
 33         options:
 34         {
 35             purge: "Delete database(s)"
 36             ,
 37             sync: "Write database(s) to file"
 38             ,
 39             backup: "Back-up databases(s)"
 40             ,
 41             all: "Apply actions to all open databases as well"
 42         }
 43         ,
 44         perm: function (src)
 45         {
 46             return sys.auth(src) == 3;
 47         }
 48         ,
 49         code: function (src, cmd)
 50         {
 51             var dblist = cmd.args;
 52             var now = +new Date;
 53 
 54             if (!cmd.flags.all && !cmd.flags.sync && !cmd.flags.backup && !cmd.flags.purge && cmd.args.length == 0)
 55             {
 56                 this.com.message([src], "Loaded databases:", this.theme.INFO);
 57                 this.less.less(src, Object.keys(this.io.openDBs).join("\n"), false);
 58                 return;
 59             }
 60 
 61             
 62 
 63             if (cmd.flags.all)
 64             {
 65                 var dblist2 = Object.keys(this.io.openDBs);
 66                 for (var x in dblist2) if (dblist.indexOf(dblist2[x]) === -1) dblist.push(dblist2[x]);
 67             }
 68 
 69             if (cmd.flags.sync) for (var x in dblist) 
 70             {
 71                 var start = +new Date;
 72                 this.io.flushDB(dblist[x]);
 73                 var end = +new Date;
 74                
 75             }
 76 
 77             if (cmd.flags.commit) for (var x in dblist) 
 78             {
 79                 var start = +new Date;
 80                 this.io.commitDB(dblist[x]);
 81                 var end = +new Date;
 82                 
 83             }
 84 
 85             if (cmd.flags.backup) for (var x in dblist) 
 86             {
 87                 var start = +new Date;
 88                 this.io.backupDB(dblist[x]);
 89                 var end = +new Date;
 90                 
 91             }
 92 
 93             if (cmd.flags.purge) for (var x in dblist)
 94             {
 95                 this.io.purgeDB(dblist[x]);
 96             }
 97         }
 98     }
 99 });
