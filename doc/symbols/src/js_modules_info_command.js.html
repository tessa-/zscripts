 22 ({
 23     require: ["profile", "com", "text", "commands", "theme", "logs"]
 24 
 25     ,
 26 
 27     info:
 28     {
 29         desc: "Shows information about users"
 30         ,
 31         perm: function (src)
 32         {
 33             return sys.auth(src) >= 1;
 34         }
 35         ,
 36         code: function (src, cmd)
 37         {
 38             var profids = [];
 39             for (var x in cmd.args)
 40             {
 41                 var test = this.profile.profileByName(cmd.args[x]);
 42                 if (test == -1) test = this.profile.profileByIP(cmd.args[x]);
 43                 
 44                 if (test == -1) 
 45                 {
 46                     this.com.message([src], "Can't find user " + cmd.args[x], this.theme.WARN);
 47                     continue;
 48                 }
 49                 else if (test === undefined || test === "" || test === null)
 50                 {
 51                     this.com.broadcast("ZOMMMG W.T.F.", this.theme.CRITICAL);
 52                     script.log(JSON.stringify(this.profile.database));
 53                     script.log(JSON.stringify(this.profile.relationaldatabase));
 54 
 55                     script.log(this.profile.profileByName(cmd.args[x]));
 56                     script.log(this.profile.profileByIP(cmd.args[x]));
 57                     continue;
 58                 }
 59                 
 60                 profids.push(test);
 61             }
 62 
 63             if (profids.length == 0) return;
 64             
 65             var m = [];
 66             for (var x in profids)
 67             {
 68                 try {
 69                     m.push("<b>Info about profile #" + this.text.escapeHTML(""+profids[x]) + ":</b>");
 70                 } catch (e)
 71                 {
 72                     this.logs.logMessage(this.logs.ERROR, "ERR " + typeof profids[x]);
 73                     this.logs.logMessage(this.logs.ERROR, "" + profids[x]);
 74                     throw e;
 75                 }
 76 
 77                 m.push("Last used name: " + this.text.escapeHTML(""+this.profile.database.profiles[profids[x]].lastName));
 78                 m.push("Last used IP: " + this.text.escapeHTML(""+this.profile.database.profiles[profids[x]].lastIP));
 79                 m.push("Names: " + this.text.escapeHTML(""+JSON.stringify(this.profile.profileNames(profids[x]))));
 80                 m.push("IP Addresses: " + this.text.escapeHTML(""+JSON.stringify(this.profile.profileIPs(profids[x]))));                       
 81             }
 82 
 83             this.com.message([src], "<br/>"+m.join("<br/>"), this.theme.INFO, true);
 84         }
 85     }
 86     ,
 87     loadModule: function()
 88     {
 89         this.commands.registerCommand("info", this);
 90     }
 91 });
