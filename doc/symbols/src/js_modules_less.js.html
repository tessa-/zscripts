 22 /** Provides the less function
 23  * @name less
 24  * @memberOf script.modules
 25  * @namespace
 26  * */
 27 /** @scope script.modules.less */
 28 ({
 29     require: ["chat", "com", "theme"]
 30     ,
 31     /** Less function
 32      * @param {Number} src User to send less to
 33      * @param msg The message to send
 34      * @param html If msg is html.
 35      */
 36     less: function (src, msg, html)
 37     {
 38         var lines;
 39 
 40         if (!html) lines = msg.split(/\n/g);
 41         else lines = msg.split(/<\s*br\s*\/?>/g);
 42 
 43         var p = 0;
 44 
 45         var bind = this;
 46 
 47         function showPage()
 48         {
 49 
 50             for (var i = 0; i < 20 && i+20*p < lines.length; i++)
 51             {
 52                 bind.com.message([src], lines[i+20*p], -1, html);
 53             }
 54             p++;
 55 
 56             if (p*20 >= lines.length)
 57             {
 58                 bind.com.message([src], "<span style='color:white;background-color:black'>END OF TEXT.</span>", -1, true);
 59                 return;
 60             }
 61 
 62             bind.com.message([src], "<span style='color:white;background-color:black'>END OF PAGE " + (p) +". TYPE 'NEXT' TO GO FORWARD, 'EXIT' TO QUIT.</span>", -1, true);
 63 
 64             bind.chat.registerCapture(src, handle, bind);
 65 
 66         }
 67 
 68         function handle (_, msg)
 69         {
 70             if (msg.toLowerCase() == "exit") return;
 71 
 72             if (msg.toLowerCase() == "next" || msg.toLowerCase() == "n")
 73             {
 74                 showPage();
 75                 return;
 76             }
 77 
 78             bind.com.message([src], "<span style='color:white;background-color:black'>UNKNOWN COMMAND.</span>", -1, true);
 79             bind.chat.registerCapture(src, handle, bind);
 80         }
 81 
 82         showPage();
 83     }
 84 });
