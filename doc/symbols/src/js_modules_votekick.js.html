 22 ({
 23 
 24     require: ["commands", "reputation", "theme", "com"]
 25     ,
 26     loadModule: function ()
 27     {
 28         this.commands.registerCommand("votekick", this);
 29         this.commands.registerCommand("yes", this);
 30         this.commands.registerCommand("no", this);
 31         this.commands.registerCommand("endvote", this);
 32     }
 33     ,
 34     votedb: null
 35     ,
 36     votekicktarget: null
 37     ,
 38     /*
 39     startvotekick: function (target)
 40     {
 41         votedb = new Object;
 42         votekicktarget = target;
 43     }
 44     */
 45     //,
 46     votekick:
 47     {
 48         perm: function ()
 49         {
 50             return true;
 51         }
 52         ,
 53         code: function (src, cmd, chan)
 54         {
 55             if (cmd.args.length != 1) return;
 56 
 57 
 58             var id = sys.id (cmd.args[0]);
 59 
 60             if (!id)
 61             {
 62                 this.com.message([src], "Invalid username", this.theme.WARN);
 63                 return;
 64             }
 65 
 66             this.com.broadcast(sys.name(src) + " calls votekick against " + cmd.args[0] + "! Type /yes or /no .", this.theme.CRITICAL);
 67 
 68             this.votedb = new Object;
 69 
 70             this.votekicktarget = id;
 71 
 72             this.votedb[src] = true;
 73         }
 74     }
 75     ,
 76     yes:
 77     {
 78         perm: function () { return true; }
 79         ,
 80         code: function (src, cmd, chan)
 81         {
 82             if (!this.votedb) return;
 83             this.votedb[src] = true;
 84         }
 85     }
 86     ,
 87     no: 
 88     {
 89         perm: function () { return true; }
 90         ,
 91         code: function (src, cmd, chan)
 92         {
 93             if (!this.votedb) return;
 94             this.votedb[src] = false;
 95         }
 96     }
 97     ,
 98     endvote: 
 99     {
100         perm: function () { return false; }
101         ,
102         code: function (src, cmd, chan)
103         {
104             if (this.votedb === null) return;
105             var yesV = 0;
106             var noV = 0;
107             var ipDB = new Object;
108             for (var x in this.votedb)
109             {
110 
111                 if (! sys.name(x)) continue;
112 
113                 if (!ipDB[sys.ip(x)]) ipDB[sys.ip(x)] = 0;
114 
115                 ipDB[sys.ip(x)] += (this.votedb[x]?1:-1) * Math.max(0,this.reputation.reputationOf(x) - (sys.connections(sys.ip(x)) - 1) * 50);
116                                 
117             }
118 
119             for (var x in ipDB)
120             {
121                 if (ipDB[x] >= 0)
122                 {
123                     yesV += Math.log(Math.E + ipDB[x]);
124                 }
125                 else
126                 {
127                     noV += Math.log(Math.E + Math.abs(ipDB[x]));
128                 }
129             }
130 
131             this.com.broadcast("Weighted vote totals: (yes points: " + yesV + ") (no points: " + noV + ")");
132 
133             this.votedb = null;
134         }
135     }
136 });
