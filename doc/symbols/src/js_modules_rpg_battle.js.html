 22 ({
 23     battleStep: function (ctx)
 24     {
 25         var rpg = ctx.rpg;
 26         var id = ctx.battleid;
 27 
 28         var battle = ctx.battle = this.battles[id];
 29 
 30 
 31 	/*
 32 	  A battle object might look like this:
 33 
 34 	  {
 35 	      players: [ "Player1", "Player2" ]
 36 	      ,
 37 	      mobs: [ {... mob ...} ]
 38 
 39 	  }
 40 
 41 	  // Turns are atomic, so there is no turn information
 42 
 43 	  
 44 	 */
 45 
 46         var team_players = []; // Do not save!
 47         for (var x in battle.players)
 48         {
 49             team_players.push(this.players[battle.players[x]]);
 50             // Battle players contains NAMES of players, not the player objects!
 51         }
 52 
 53         var team_mobs = []; // Do not save!
 54         for (var x in battle.mobs)
 55         {
 56             team_mobs.push(battle.mobs[x]);
 57         }
 58         
 59         var entities = []; // Do not save!
 60 
 61 
 62         for (var x in team_players)
 63         {
 64             entities.push({type: "player", e: team_players[x]});
 65         }
 66 
 67         for (var x in team_players)
 68         {
 69             entities.push({type: "mob", e: team_mobs[x]});
 70         }
 71 
 72         entities.sort( 
 73             function _sorting_function_ (a, b)
 74             {
 75                 return a.speed - b.speed;
 76             }
 77         );
 78 
 79         battleLoop: for (var x in entities)
 80         {
 81             var attacker = ctx.attacker = entities[x].e;
 82             //ctx.attackerIsPlayer = entities[x].type == "player";
 83             var move = ctx.move = this.pickMove(entities[x]);
 84 
 85             if (ctx.move.cost) for (var x2 in ctx.move.cost)
 86             {
 87                 if (ctx.attacker[(x2 === "mp"? "mana" : x)] -= ctx.move.cost[x2] < 0) 
 88                 {
 89                     ctx.out(ctx.attacker.name + " tried to use "  + ctx.move.name + " but didn't have enough " + this.longStatName[x2 === "mp" ? "mana" : "mp"]);
 90                     continue battleLoop;
 91                 }           
 92             }
 93             
 94             for (var x2 in ctx.move.components)
 95             {
 96                 var cmp = ctx.move.components[x2];
 97                 var targets = ctx.targets = [];
 98                 var count = cmp.count;
 99                 switch (typeof cmp.target)
100                 {
101                 case "object":
102                     // Array
103                     if (cmp.target.indexOf("ally") != -1) ctx.targets = ctx.targets.concat(team_players);
104                     if (cmp.target.indexOf("opp") != -1) ctx.targets = ctx.targets.concat(team_mobs);
105                     if (cmp.target.indexOf("self") != -1 && ctx.indexOf(ctx.attacker) == -1) ctx.targets.push(ctx.attacker);
106                     break;
107                 case "string":
108                     switch (ctx.move.type)
109                     {
110                         case "ally": ctx.targets = team_players;
111                         case "self": ctx.targets = [attacker];
112                     }
113                 }
114 
115                 this.util.shuffle(targets);
116                 
117                 if (count !== 0) for (var x3 in targets)
118                 {
119                     if (count-- === 0) break;
120                 }                
121             }
122             
123             this.moves[ctx.move.type]
124         }
125         
126     }
127     
128 });
