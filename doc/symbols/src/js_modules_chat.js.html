 22 (function () { return {
 23     require: ["commands", "security", "profile", "com", "theme", "time", "logs", "text", "util"]
 24     ,
 25     filters: []
 26     ,
 27     capture: new Object
 28     ,
 29     registerCapture: function (src, func, object)
 30     {
 31         if (!this.capture[src]) this.capture[src] = [];
 32 
 33         var capper = this.util.bind(object, func);
 34 
 35         this.capture[src].push(capper);
 36         object.onUnloadModule( this.util.bind(
 37             this,
 38             function ()
 39             {
 40                 this.filters.splice(this.filters.indexOf(capper), 1);
 41             }
 42         ));
 43 
 44         // sys.sendHtmlMessage(src, "<span style=\"background-color:#000000; color:white\"> INTERACTIVE MODE ACTIVE </span>");
 45     }
 46     ,
 47     beforeChatMessage: function beforeChatMessage (src, msg, chan)
 48     {
 49         if (msg.length == 0) return;
 50 
 51         if (this.capture[src] && this.capture[src].length >= 1)
 52         {
 53             sys.sendHtmlMessage(src, "<span style=\"background-color:#000000; color:white\">:"+ this.text.escapeHTML(msg));
 54             sys.stopEvent();
 55             var f = this.capture[src].pop();
 56 
 57             f.call(new Object, src, msg, chan);
 58 
 59             return;
 60         }
 61         
 62         if (msg[0] == "/")
 63         {
 64             this.logs.logMessage(this.logs.USER, "" + sys.name(src) + "["+chan+"] " + msg);
 65             sys.stopEvent();
 66             this.commands.issueCommand(src, msg, chan);
 67             return;
 68         }
 69 
 70         var prof = this.profile.profileID(src);
 71         
 72         if (sys.auth(src) != 3 && this.security.profIsMuted(prof))
 73         {
 74             var mute = this.security.getMute(prof);
 75             
 76             this.com.message(
 77                 [src],
 78                 "You are muted. "+
 79                     (mute.expires ? new Date(mute.expires).toString() + " ("+this.time.diffToStr(mute.expires - +new Date)  +" from now)" : "indefinite" )+
 80                     " Reason: " + mute.reason +
 81                     " Mute Author: " + mute.author
 82                 ,
 83                 this.theme.WARN
 84             );
 85             
 86             this.logs.logMessage(this.logs.INFO, "Muted message from "+sys.name(src)+": " + msg);
 87             sys.stopEvent();
 88             
 89             return;
 90         }
 91 
 92         var m = msg;
 93 
 94         for (var x in this.filters)
 95         {
 96             m = this.filters[x](src, msg, chan);
 97 
 98             if (!m) break;
 99         }
100 
101         if (m) 
102         {
103             sys.broadcast(m, chan, src, false, -1);
104             //this.com.broadcast("<timestamp /><b>" +sys.name(src) + ":</b> " + this.text.escapeHTML(m), -1, true, [chan]);
105         }
106         sys.stopEvent();
107         
108     }
109     ,
110     registerFilter: function (filter, object)
111     {
112         var filt = filter;
113         this.filters.push(filt);
114         object.onUnloadModule( this.util.bind(
115             this,
116             function ()
117             {
118                 this.filters.splice(this.filters.indexOf(filt), 1);
119             }
120         ));
121     }
122     ,
123     loadModule: function loadModule ()
124     {
125         script.registerHandler("beforeChatMessage", this);
126     }
127 }})();
