 30         walk: function (src, sub, chan, ctx)
 31         {
 32             for (var i = 1; i < sub.length; i++)
 33             {                
 34                 var to = sub[i];
 35 
 36                 if (this.areas[ctx.player.area].adjc.indexOf(to) !== -1)
 37                 {
 38 
 39                     this.com.message([src], "Walked from " + this.areas[ctx.player.area].name + " to " + this.areas[to].name + ".", this.theme.GAME);
 40 
 41                     ctx.player.area = to;
 42                 }
 43                 else
 44                 {
 45                     this.com.message([src], "Can't find that place!", this.theme.GAME);
 46                 }
 47             }
 48 
 49             this.com.message([src], "You are at: " + this.areas[ctx.player.area].name + ". From here you can go to:", this.theme.GAME, false, [chan]);
 50 
 51 
 52             
 53             for (var i = 0; i < this.areas[ctx.player.area].adjc.length; i++)
 54             {
 55                 this.com.message([src], this.areas[this.areas[ctx.player.area].adjc[i]].name + " (" + this.areas[ctx.player.area].adjc[i] + ")", -1, false, chan);
 56             }        
 57 
 58             
 59         }

 61         dig: function (src, sub, chan, ctx)
 62         {
 63             ctx.player.activeActions.push({ timer: 40, done: "dig", tick: "digTick" });
 64         }

 66         dequip: function (src, sub, chan, ctx)
 67         {
 68             var slot = (sub[1]||"").toLowerCase();
 69             
 70             if (! (slot in {"lhand":null, "rhand":null, "head":null, "feet":null, "body":null, "back":null})) 
 71             {
 72                 this.com.message([src], "Unknown slot to dequip");
 73                 return; 
 74             }// add error message
 75 
 76             var item = ctx.player[slot];
 77 
 78             if (item === undefined) throw new Error("???");
 79 
 80             if (item === null && slot === "lhand" && ctx.player.rhand && ctx.player.rhand.hands === 2)
 81             {
 82                 item = ctx.player.rhand;
 83                 slot = "rhand";
 84             }
 85 
 86             if (item === null) 
 87             {
 88                 this.com.message(src, "No item in that slot.");
 89                 return; // nothing to dequip
 90             }
 91 
 92             ctx.player[slot] = null; // remove item
 93 
 94             ctx.player.equips.unshift(item); // Add to equips
 95 
 96             this.com.message([src], "Item removed.");
 97             
 98         }
 99         ,
100         view: function (src, sub, chan, ctx)
101         {
102             var msgs = [];
103 
104             this.com.message(src, "Your player:");
105             if (ctx.player.rhand && this.equips[ctx.player.rhand.type].hands === 2)
106             {
107                  msgs.push("<b>Both Hands:</b> " + this.equipName(ctx.player.rhand));
108             }
109             else
110             {
111                 msgs.push("<b>Right Hand:</b> " + this.equipName(ctx.player.rhand));
112                 msgs.push("<b>Left Hand:</b> " + this.equipName(ctx.player.lhand));
113             }
114             msgs.push("<b>Head:</b> " + this.equipName(ctx.player.head));
115             msgs.push("<b>Body:</b> " + this.equipName(ctx.player.body));
116             msgs.push("<b>Feet:</b> " + this.equipName(ctx.player.feet));
117             msgs.push("<b>Back:</b> " + this.equipName(ctx.player.back));
118             msgs.push("<b>Power:</b> " + ctx.player.power);
119             msgs.push("<b>Offense:</b> " + ctx.player.offense);
120             msgs.push("<b>Defense:</b> " + ctx.player.defense);
121             msgs.push("<b>Magical:</b> " + 0/0);
122 
123             this.less.less(src, msgs.join("<br />"), true);
124         }
125         ,
126         equip: function (src, sub, chan, ctx)
127         {
128             if (!sub[1])
129                 // list equips
130             {
131                 for (var x in ctx.player.equips)
132                 {
133                     this.com.message(src, "" + (+x+1) + ": " + this.equipName(ctx.player.equips[x]));
134                 }
135                 
136                 return;
137             }
138 
139             var idx = +sub[1] - 1;
140 
141             var item = ctx.player.equips.splice(idx, 1)[0];
142             var kind = this.equips[item.type];
143 
144             if (!item) return; // invalid indexing ._.
145 
146             var slot = kind.type;
147 
148             if (slot === "hand") 
149             {
150                 if (sub[2] === "lhand") slot = "lhand";
151                 else if (sub[2] === "rhand") slot = "rhand";
152                 else
153                 {
154                     if (kind.hands !== 2)
155                         // error
156                     {
157                         ctx.player.equips.unshift(item); // put item back
158                         this.com.message(src, "Which hand?");
159                         return; // exit
160                     }
161 
162                     slot = "rhand";
163 
164                     if (ctx.player.lhand)
165 
166                         // remove left hand equip
167                     {
168                         ctx.player.equips.unshift(ctx.player.lhand);
169                         ctx.player.lhand = null;
170                     }
171                 }
172             }
173 
174             if (ctx.player[slot]) ctx.player.equips.unshift(ctx.player[slot]);// put old item back
175 
176             if (slot === "lhand" && ctx.player.rhand && this.equips[ctx.player.rhand.type].hands === 2)
177             {
178                 ctx.player.equips.unshift(ctx.player.rhand);// put old item back
179                 ctx.player.rhand = null;
180             }
181 
182             ctx.player[slot] = item;
183 
184             this.com.message(src, "Equipped " + this.equipName(item));
185 
186             return;
187         }
188
