 22 /** Contains the RPG game :)
 23  * @name rpg_game
 24  * @memberOf script.modules
 25  * @requires io, com, theme, commands, util, logs, less
 26  * @namespace
 27  * */
 28 (
 29 /** @scope script.modules.rpg_game */
 30 {
 31     /** List of modules to be required.
 32      * @type {string[]}
 33      */
 34     require: ["io", "com", "theme", "commands", "util", "logs", "less"]
 35     ,
 36     /** List of files to be included in this module.
 37      * @type {string[]}
 38      */
 39     include: ["rpg_areas", "rpg_player", "rpg_entity", "rpg_actions",
 40               "rpg_mobs", "rpg_equips", "rpg_materials", "rpg_moves"]
 41     ,
 42     /** I/O database that keeps track of games etc.
 43      * @type {IOdatabase}
 44      */
 45     database: null
 46     // database stores the permanent data, games, etc.
 47     ,
 48     /** */
 49     channels: null
 50     // channels is used to associate databases with channels, it doesn't keep between reloads
 51     ,
 52     hooks: null
 53     // hooks associates rpgs with pipes
 54     // not used as it should be for right now, will be fixed later
 55     ,
 56     loadModule: function ()
 57     {
 58         this.database = this.io.openDB("rpg_game");
 59         if (!this.database.games) this.database.games = new Object;
 60         this.channels = new Object;
 61         this.hooks = new Object;
 62 
 63         this.commands.registerCommand("loadrpg", this);
 64         this.commands.registerCommand("rpg", this);
 65 
 66         script.registerHandler("step", this);
 67     }
 68     ,
 69     unloadModule: function ()
 70     {
 71         this.io.closeDB("rpg_game");
 72     }
 73     ,
 74     step: function ()
 75     {
 76         for (var x in this.channels)
 77         {
 78             this.RPGStep(this.channels[x]);
 79         }
 80     }
 81     ,
 82     RPGStep: function (rpg)
 83     {
 84         if (rpg.paused) return;
 85         rpg.tick++;
 86 
 87         for (var x in rpg.areas)
 88         {
 89             this.areaStep(rpg.areas[x], { rpg: rpg });
 90         }
 91 
 92         for (x in rpg.players)
 93         {
 94             this.playerStep(rpg.players[x], {rpg: rpg});
 95         }
 96     }
 97     ,
 98     RPG: function (rpgname)
 99     {
100         if (this.database.games[rpgname]) { return this.database.games[rpgname]; }
101 
102         else
103         {
104             var newr = this.database.games[rpgname] =
105                 {
106                     name: rpgname,
107                     areas: JSON.parse(JSON.stringify(this.areas)),
108                     materials: {},
109                     players: {},
110                     battles: {},
111                     battleCoutner: 0,
112                     running: false,
113                     tick: 0
114                 };
115 
116             for (var x in newr.areas)
117             {
118                 newr.areas[x].battles = [];
119             }
120         }
121 
122         return newr;
123     }
124     ,
125     startRPGinChan: function (rpgname, chan)
126     {
127         var rpg = this.database.games[rpgname];
128 
129         this.channels[chan] = rpg;
130 
131         if ( this.hooks[rpgname] ) throw new Error("Already hooked into another channel?");
132 
133         this.logs.logMessage(this.logs.INFO, "Loaded RPG game " + rpgname + " in channel " + sys.channel(chan));
134 
135         this.hooks[rpgname] =
136             {
137                 message: this.util.bind
138                 (
139                     this
140                     ,
141                     function(player, message)
142                     {
143                         this.com.message([player], message, this.theme.GAME, true, [chan]);
144                     }
145                 )
146                 ,
147                 broadcast: this.util.bind
148                 (
149                     this
150                     ,
151                     function (message)
152                     {
153                         this.com.broadcast(message, this.theme.GAME, true, [chan]);
154                     }
155                 )
156             };
157 
158 
159         this.hooks[rpgname].running = true;
160 
161     }
162     ,
163     /** @type commandDescriptor */
164     rpg: // not to be confused with "RPG"
165     {
166         desc: "does rpg stuffs",
167         perm: function ()
168         {
169             return true;
170         }
171         ,
172         code: function (src, cmd, chan)
173         {
174             if (!this.channels[chan])
175             {
176                 this.com.message([src], "No RPG running in that channel", this.theme.WARN);
177                 return;
178             }
179 
180             var rpg = this.channels[chan];
181 
182             var player = rpg.players[sys.name(src).toLowerCase()];
183             if (! player)
184             {
185                 this.com.message([src], "Creating you a new RPG character!", this.theme.GAME);
186                 this.logs.logMessage(this.logs.INFO, sys.name(src) + " created an RPG character in RPG " + rpg.name);
187 
188                 player = this.newPlayer();
189                 player.name = sys.name(src).toLowerCase();
190 
191                 rpg.players[sys.name(src).toLowerCase()] = player;
192             }
193 
194             var actions = String(cmd.input).split(/\;/g);
195 
196             for (var x in actions)
197             {
198                 var subactions = actions[x].split(/[,:]/g);
199 
200                 if (subactions.length == 0) continue;
201 
202                 subactions[0] = subactions[0].toLowerCase();
203                 if (subactions[0] in this.rpgActions)
204                 {
205                     this.rpgActions[subactions[0]].apply(this, [src, subactions, chan, {player:player, rpg: rpg }] );
206                 }
207             }
208         }
209     }
210     ,
211     /** @type commandDescriptor */
212     loadrpg:
213     {
214         desc: "Loads an RPG into a channel."
215         ,
216         perm: function (src)
217         {
218             return sys.auth(src) >= 2;
219         }
220         ,
221         code: function (src, cmd, chan)
222         {
223 
224             var rpgname = cmd.input.replace(/^\s*([^\s]+)\s*$/, "$1");
225 
226             if (this.channels[chan])
227             {
228                 this.com.message([src], "This channel already has a running RPG", this.theme.WARN);
229                 return;
230             }
231 
232             var rpg = this.RPG(rpgname);
233             if (this.hooks[rpgname])
234             {
235                 this.com.message([src], "RPG is already running!", this.theme.WARN);
236                 return;
237             }
238 
239             this.startRPGinChan(rpgname, chan);
240         }
241     }
242 
243 });
